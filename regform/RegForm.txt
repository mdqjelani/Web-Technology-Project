        function validatePassword() {
            var p = document.getElementById('myInput').value,
            errors = [];
        if (p.length < 8) {
            errors.push("must be at least 8 characters"); 
        }
        if (p.search(/[a-z]/i) < 0) {
            errors.push("must contain at least one letter.");
        }
        if (p.search(/[0-9]/) < 0) {
            errors.push("must contain at least one digit."); 
        }
        if (errors.length > 0) {
            alert(errors.join("\n"));
            return false;
        }
        return true;
